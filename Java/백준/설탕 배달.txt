#include <stdio.h>

int main(void){
int N;
//5로 나눠질땐 그냥 5의 몫을 출력, 안나눠질 때
scanf("%d", &N);

if(N==0) printf("-1");	
else if(N%3!=0&&N%5!=0){
	if(N<8) printf("-1"); 
    else if(N%5==1)	printf("%d", (N/5)+1);	
	else if(N%5==2)	printf("%d", (N/5)+2);
	else if(N%5==3)	printf("%d", (N/5)+1);
	else if(N%5==4)	printf("%d", (N/5)+2);
	else printf("-1");
}
else if(N%3==0&&N%5!=0){
	if(N<10) printf("%d", N/3);
	else if(N%5==1)	printf("%d", (N/5)+1);	
	else if(N%5==2)	printf("%d", (N/5)+2);
	else if(N%5==3)	printf("%d", (N/5)+1);
	else if(N%5==4)	printf("%d", (N/5)+2);
    else printf("%d", N/3);
}
else if(N%3!=0&&N%5==0){
		printf("%d", N/5);
}
else if(N%3==0&&N%5==0){
		printf("%d", N/5);

}

return 0;
}
